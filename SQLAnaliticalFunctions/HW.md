# Q1
```sql
SELECT DEPARTMENT_ID, LISTAGG(FIRST_NAME||' '||LAST_NAME,';') WITHIN GROUP(ORDER BY DEPARTMENT_ID) AS sonuc FROM HR.EMPLOYEES GROUP BY DEPARTMENT_ID 
```

# Q2
```sql
select HR.EMPLOYEES.*, SUM(SALARY) OVER (PARTITION BY JOB_ID ORDER BY EMPLOYEE_ID ROWS BETWEEN 1 PRECEDING AND 1 FOLLOWING) NUM from HR.EMPLOYEES
```

# Q3
```sql
select HR.EMPLOYEES.*, LEAD(PHONE_NUMBER,1) OVER (PARTITION BY DEPARTMENT_ID ORDER BY HIRE_DATE) NEXT_VALUE  from HR.EMPLOYEES order by HIRE_DATE
```

# Q4
```sql
select HR.EMPLOYEES.*, DENSE_RANK() OVER(ORDER BY SALARY,HIRE_DATE DESC) NUM FROM HR.EMPLOYEES 
```

# Q5
```sql
select HR.EMPLOYEES.*, NTILE(11) OVER (ORDER BY EMPLOYEE_ID) from HR.EMPLOYEES
```

# Q6
```sql
SELECT DEPARTMENT_ID, SALARY, AVG(SALARY) OVER (PARTITION BY DEPARTMENT_ID),
    (CASE
        WHEN (SALARY < (SELECT AVG(B.SALARY) FROM HR.EMPLOYEES B WHERE A.DEPARTMENT_ID = B.DEPARTMENT_ID))
        THEN '0' ELSE '1' END) salary
FROM HR.EMPLOYEES A
```
